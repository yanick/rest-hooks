PollingSubscription implements Subscription

Will dispatch a fetch action at the minimum interval of all subscriptions to this resource.

import { SubscriptionManager, PollingSubscription, RestProvider } from 'rest-hooks';
import ReactDOM from 'react-dom';

const subscriptionManager = new SubscriptionManager(PollingSubscription);

ReactDOM.render(
  <RestProvider subscriptionManager={subscriptionManager}>
    <App />
  </RestProvider>,
  document.body
);

This is already used by RestProvider by default.
